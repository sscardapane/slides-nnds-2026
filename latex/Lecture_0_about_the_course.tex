\begin{frame}{Learning objectives}


	\begin{itemize}
		\item Broad historical overview of the last 15 years.
		\item Fundamental tools underlying neural networks: optimization, gradient descent, automatic differentiation.
		\item Basic blocks to build modern neural networks (convolution, attention, normalization, ...).
		\item Proficiency in a real-world deep learning library (\textbf{JAX}, \textbf{PyTorch}).
		\item Capability of navigating the current literature and ecosystem in autonomy.
	\end{itemize}

\end{frame}

\begin{frame}{Lectures (methodological)}

	\begin{enumerate}
		\item \textbf{Historical overview}.
		\item \textbf{Preliminaries} (tensors, linear algebra, optimization).
		\item \textbf{Supervised learning} as numerical optimization.
		\item Linear models and \textbf{fully-connected} models.
		\item \textbf{Convolutional models} (for sequential, spatial, and temporal data).
		\item Blocks to train \textbf{deeper models} (dropout, batch normalization, ...).
		\item \textbf{Attention} models for sets.
		\item \textbf{LLM} building blocks: autoregression, tokenization, pre- and post-training.
		\item \textbf{Graph} models (e.g., graph convolutional networks).
		\item Optional topics depending on time and material.
	\end{enumerate}

\end{frame}

\begin{frame}{Lectures (practical)}

	\begin{enumerate}
		\item Several practical lectures with \textbf{JAX} and \textbf{PyTorch} (hands-on coding from scratch).
		\item When possible, a showcase of other libraries (e.g., HuggingFace Datasets).
		\item Optional topics depending on time and material.
	\end{enumerate}

\end{frame}
